// 7 RVR
// Robot 1 to 7, left to right.
// The distance for running 50 for 2 second is 65
// The box is a 65cm quare
// The distance between boxes can be 20cm
// Which is 50 for 0.3 second

async function giftBox(){
	await roll(0,50,2.0)
	await stopRoll()
	await delay (0.2)
	await roll(90,50,2.0)
	await stopRoll()
	await delay (0.2)
	await roll(180,50,2.0)
	await stopRoll()
	await delay (0.2)
	await roll(270,50,2.0)
	await stopRoll()
	await delay (0.2)
}
